# 插槽 {props.children}

```JavaScript
// App.tsx
function Tags() {
  return (
    <PageLayout>
      <h2>Tags</h2>
    </PageLayout>
  );
}
```

```JavaScript
// PageLayout.tsx
const PageLayout = (props: any) => {
   // 这里只是暂时写的 any 后面要改
   return (
    <Wrapper>
      <Main>
        {props.children}
      </Main>
      <Nav/>
    </Wrapper>
   )
};
// <h2>Tags</h2> 就会被放到 PageLayout 组件中的 {props.chilren} 里去。
```
